#include <stdio.h>
#include <string.h>
#include <stdlib.h>


typedef int Type;

typedef struct {
  
    Type* data;     // vetor que armazenará os elementos;
    int  size;     // tamanho atual do vetor
    int  capacity; // tamanho reservado em memória
  
} vector;

// Alocação/desalocação de memória
vector* new_vector(int initial_capacity)
{
    vector*v = (vector*)malloc(sizeof(vector));
  
    v->data = (int*)malloc(sizeof(int)*initial_capacity);//tem que ser a capacidade inicial, pois ele ainda nao
    //tinha definido.  
    v->size = 0;
    v->capacity = initial_capacity;
  
    return v;      
}
void free_vector(vector* v)
{
    int i;
      
    free(v->data);

    free(v);
}

// Funções de remoção de elementos
void vector_pop_back(vector* v)
{
    v->data[v->size-1] = 0;
    v->size--;
   
}

void vector_pop_front(vector* v)
{
    int i;
      

    for(i=0;i<v->size;i++)
    {
        v->data[i]=v->data[i+1];
    }     
    v->size--;
}

void vector_erase(vector* v, int i)
{  
    int j;
    v->data[i] = 0;
    
    for(j=i;j<v->size;j++)
    {
        v->data[j] = v->data[j+1];
    }      
    v->size--;
}

// Impressão do valores no vetor [0]...[size-1]
void vector_print(vector* v)
{

    printf("[%d/%d]",v->size,v->capacity);
    
    for(i=0;i<v->size;i++)
    {
        printf(" %d",v->data[i]);

}
