#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

#include <assert.h>



typedef int Type;

typedef struct {
    Type* data;
    int  size;
    int  capacity;
} vector;


vector* new_vector(int initial_capacity)
{
    vector* v = malloc(sizeof(vector));
    v->size = 0;
    v->capacity = initial_capacity;
    if(initial_capacity==0)
    {
        v->capacity = 4;
    }
    v->data = malloc(sizeof(int)*v->capacity);
    return v;
}


void vector_resize(vector* v)
{
    v->capacity = v->capacity*2;
    v->data = realloc(v->data,sizeof(int)*v->capacity);
}


void free_vector(vector* v)
{
    free(v);
}


void vector_push_back(vector* v, Type value)
{
    if(v->size == v->capacity)
    {
        vector_resize(v);
    }
    v->data[v->size] = value;
    v->size++;
}


void vector_push_front(vector* v, Type value)
{
    if(v->size == v->capacity)
    {
        vector_resize(v);
    }
    int i=v->size;
    for(i; i>0; i--)
    {
        v->data[i] = v->data[i-1];
    }
    v->data[0] = value;
    v->size++;
}


void vector_insert(vector* v, Type value, int i)
{
    if(v->size == v->capacity)
    {
        vector_resize(v);
    }
    int j=v->size;
    for(j; j>i; j--)
    {
        v->data[j] = v->data[j-1];
    }
    v->data[i] = value;
    v->size++;
}


Type vector_pop_back(vector* v)
{
    assert(v->size>0);
    v->size--;
    return v->data[v->size];
}


Type vector_pop_front(vector* v)
{
    assert(v->size>0);
    Type ret = v->data[0];
    int i =0 ;
    int j = v->size;
    for(i; i<j; i++)
    {
        v->data[i] = v->data[i+1];
    }
    v->size--;
    return ret;
}


Type vector_erase(vector* v, int i)
{
    assert(v->size>0);
    Type ret = v->data[i];

    int j = v->size;
    for(i; i<j; i++)
    {
        v->data[i] = v->data[i+1];
    }
    v->size--;
    return ret;
}


void vector_set(vector* v, int pos, Type value)
{
    v->data[pos] = value;
}


void vector_print(vector* v)
{
    printf("[%d/%d] ", v->size, v->capacity);
    int j = v->size;
    for(int i = 0; i<j-1; i++)
    {
        printf("%d ",v->data[i]);
    }
    printf("%d\n",v->data[j-1]);
    //[size/capacity] data[0] data[1] data[2] ... data[size] \n
}


int main() 
{
    int cap=0, ops=0, cont=0, aux1=0, aux2=0;
    char input[15];
    scanf("%d", &cap);
    scanf("%d", &ops);
    vector* v = new_vector(cap);
    for(cont; cont<ops; cont++)
    {
        fflush(stdin);
        fflush(stdout);
        scanf("%s %d %d", input, &aux1, &aux2);
        
        //
        
        if(strcmp(input,"push_back")==0)
        {
            vector_push_back(v, aux1);
        }
        
        else if(strcmp(input,"push_front")==0)
        {
            vector_push_front(v, aux1);
        }
        
        else if(strcmp(input,"insert")==0)
        {
            vector_insert(v, aux1, aux2);
        }
        
        else if(strcmp(input,"pop_back")==0)
        {
            vector_pop_back(v);
        }
        
        else if(strcmp(input,"pop_front")==0)
        {
            vector_pop_front(v);
        }
        
        else if(strcmp(input,"erase")==0)
        {
            vector_erase(v, aux1);
        }
        
        else if(strcmp(input,"set")==0)
        {
            vector_set(v, aux1, aux2);
        }
        
        //
        if(v->size>0)
        {
            vector_print(v);
        }
        
        else if(v->size==0)
        {
            printf("[%d/%d]\n", v->size, v->capacity);
        }
        
        
    }
    
    free_vector(v);
    return 0;
}
