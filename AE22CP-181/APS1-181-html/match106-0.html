#include<stdio.h>
#include<stdlib.h>
#include<string.h>

typedef int Type;

typedef struct {
    Type* data;     // vetor que armazenará os elementos;
    int  size;     // tamanho atual do vetor
    int  capacity; // tamanho reservado em memória
} vector;

// Alocação/desalocação de memória
vector* new_vector(int initial_capacity)
{
  vector *aux=malloc(sizeof(vector));
  aux->size=0;
  aux->capacity=initial_capacity;
  aux->data=(Type*)malloc(sizeof(Type)*aux->capacity);
  return aux;
}
void free_vector(vector* v)
{
  free(v->data);
  v->size=0;
  v->capacity=0;
}

// Funções de remoção de elementos
Type vector_pop_back(vector* v)
{
    Type aux;
    aux=v->data[v->size];
    v->size--;
    return aux;
}

Type vector_pop_front(vector* v)
{
  Type aux;
  int i,j;
  i=0;
  aux=v->data[i];
  for(j=0;j<v->size-1;j++)
    v->data[j]=v->data[j+1];
  v->size--;
  return aux;
}

Type vector_erase(vector* v, int i)
{
  Type aux;
  int j;
  aux=v->data[i];
  for(j=i;j<v->size;j++)
    v->data[j]=v->data[j+1];
  v->size--;
  return aux;
}

void vector_insert(vector* v, Type value, int i)
{
  if(v->size==v->capacity)
    vector_resize(v);
  int j;
  if(i<=v->size)
  {
  for(j=v->size;j>i;j--)
    v->data[j]=v->data[j-1];
  }
  v->data[i]=value;
  v->size++;
}

void vector_push_back(vector* v, Type value)
{
  if(v->capacity==v->size)
    vector_resize(v);
  v->size++;
  v->data[v->size-1]=value;
}

void vector_push_front(vector* v, Type value)
{
  int j;
  if(v->capacity==v->size)
    vector_resize(v);
  for(j=v->size;j>0;j--)
    v->data[j]=v->data[j-1];
  v->data[0]=value;
  v->size++;
}


// Função de alteração de uma posição do vetor
void vector_set(vector* v, int pos, Type value)
{
  v->data[pos]=value;
}

// Redimensionamento da capacidade do vetor
void vector_resize(vector* v)
{
  if(v->capacity==0)
  {
    v->capacity=4;
    v=realloc(v,sizeof(vector));
    v->data=(Type*)realloc(v->data,sizeof(Type)*v->capacity);
  }
  else
  {
    v->capacity=(v->capacity*2);
    v=realloc(v,sizeof(vector));
    v->data=(Type*)realloc(v->data,sizeof(Type)*v->capacity);
  }
}

// Impressão do valores no vetor [0]...[size-1]
void vector_print(vector* v)
{
  int i;
  i=0;
  printf("[%d/%d] ",v->size,v->capacity);
  if(v->size!=0)
  {
   while(i<v->size)
    {
      printf("%d ",v->data[i]);
      i++;
    }
  printf("\n");
  }
  else
    printf("\n");

}

int main(void)
{
  int capacity,lines,position,value;
  char operation[30];
  int i;
  Type retorno;
  scanf("%d",&capacity);
  vector *v=new_vector(capacity);
  scanf("%d",&lines);
  while(i<lines)
  {
    scanf("%s",operation);
    if(strcmp(operation,"push_back")==0)
    {
      scanf("%d",&value);
      vector_push_back(v,value);
    }
    else if(strcmp(operation,"push_front")==0)
    {
      scanf("%d",&value);
      vector_push_front(v,value);
    }
    else if(strcmp(operation,"insert")==0)
    {
      scanf("%d %d",&value,&position);
      vector_insert(v,value,position);
    }
    else if(strcmp(operation,"pop_back")==0)
    {
      retorno=vector_pop_back(v);
    }
    else if(strcmp(operation,"pop_front")==0)
    {
      retorno=vector_pop_front(v);
    }
    else if(strcmp(operation,"erase")==0)
    {
      scanf("%d",&position);
      retorno=vector_erase(v,position);
    }
    else if(strcmp(operation,"set")==0)
    {
      scanf("%d %d",&position,&value);
      vector_set(v,position,value);
    }
    vector_print(v);
    i++;
  }
  free_vector(v);
  return 0;
}
