#include <stdlib.h>
#include <stdio.h>
#include <string.h>

typedef int Type;

typedef struct{

  int* data;
  int size;
  int capacity;

} vector;

vector* new_vector(int capacidade)
{
    vector* v = malloc(sizeof(vector));
    v->capacity=capacidade;
    v->size = 0;
    v->data = malloc(sizeof(int) * capacidade);
    // calloc(sizeof(int), capacidade);
    return v;
}

// Redimensiona a capacidade do vetor
void vector_resize(vector* v)
{
    if(v->capacity==0)
    {
        v->capacity=2;        
    }
    
    v->data=realloc(v->data,sizeof(int)*(v->capacity*2));
    v->capacity=v->capacity*2;
}

// Desaloca a memória
void free_vector(vector* v)
{
    free(v);
    v=NULL;
}

// Remove o elemento do fim do vetor, decrementando size.
Type vector_pop_back(vector* v)
{
    int value;

    value=v->data[v->size-1];

    v->data[v->size-1]=0;
    v->size--;

    return value;
}
// Remove o elemento na posição i, e move os posteriores para trás
Type vector_erase(vector* v, int i)
{
    int j,value;

    value=v->data[i];

    for(j=i+1;j<v->size;j++)
    {
        v->data[j-1]=v->data[j];
    }

    v->size--;

    return value;

}

// Remove o elemento na posição zero do vetor e move os posteriores para trás
Type vector_pop_front(vector* v)
{
    int i,value;
    value=v->data[0];

    for(i=0;i<v->size;i++)
    {
      v->data[i]=v->data[i+1];
    }

    v->size--;

    return value;
}

    for(i=v->size-1;i>=0;i--)

}

{
    int j;
    
    if (v->capacity==v->size)
    {
          vector_resize(v);
    }

    for(j=v->size-1;j>=i;j--)
    {        
        v->data[j+1]=v->data[j];
    }

    v->data[i]=value;
    v->size++;

}

// Altera o valor na posição i
void vector_set(vector* v, int i, Type value)
{
    v->data[i]=value;
}

// Mostra o vetor
void vector_print(vector* v)
{
    int i;

    printf("[%d/%d] ",v->size,v->capacity);

    for(i=0;i<v->size;i++)
    {
      printf("%d ",v->data[i]);
    }

    printf("\n");

}
