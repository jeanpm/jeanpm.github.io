#include <stdio.h>
#include <assert.h>
#include <stdlib.h>
#include <string.h>


typedef int type;

typedef struct {
    type *data;     // vetor que armazenará os elementos;
    int  size;     // tamanho atual do vetor
    int  capacity; // tamanho reservado em memória
} vector;

// Alocação/desalocação de memória
vector* new_vector(int capacidade){
    vector *v = malloc(sizeof(vector));
    v->data = malloc(sizeof(int)*capacidade);
    v->size = 0;
    v->capacity = capacidade;

    return v;
}

void free_vector(vector* v){
    free(v->data);
    free(v);
}

// Redimensionamento da capacidade do vetor
void vector_resize(vector* v){
    if(v->capacity == 0){
        v->capacity = 4;
    }
    else{
        v->capacity = v->capacity * 2;
        v->data= realloc (v->data,sizeof(int) * v->capacity);
    }
}

// Funções de remoção de elementos
type vector_pop_back(vector *v){
    v->size = v->size - 1;
    return(v->data[v->size]);
}//remover ultimo elemento

type vector_pop_front(vector* v){
    for(int i=0; i < v->size; i++){
        v->data[i] = v->data[i+1];
    }
    v->size = v->size - 1;
    return(v->data[v->size]);

}//remover primeiro elemento

type vector_erase(vector* v, int i){
    assert(i >= 0 && i<=v->size);

    int x=v->data[i];
    v->size--;
    for(int j = i;j<v->size; j++)
        v->data[j]=v->data[j+1];

    v->data[v->size]=0;
    return x;
}//remove na posição selecionada;

// Funções de inserção de elementos
void vector_insert(vector* v, type value, int i){
    assert(i >= 0 && i<=v->size);
    if(v->capacity == v->size){
        vector_resize(v);
    }

    for(int j=v->size; i<j; j--){
        v->data[j]=v->data[j-1];
    }
    v->data[i] = value;
    v->size++;
}//insere na posição selecionada

void vector_push_back(vector* v, type value){
    if(v->capacity == v->size){
        vector_resize(v);
    }
    v->data[v->size] = value;
    v->size++;
}//insere no final

void vector_push_front(vector *v, type value){
    if(v->capacity == v->size){
        vector_resize(v);
    }
    for(int j=v->size; 0<j; j--){
        v->data[j]=v->data[j-1];
    }
    v->data[0] = value;
    v->size++;
}//insere no inicio

// Função de alteração de valor de uma posição do vetor
void vector_set(vector* v, int pos, type value){
    assert(pos >= 0 && v->size > pos);
    v->data[pos] = value;
}

void vector_print(vector* v){
    printf("[%d/%d] ", v->size, v->capacity);
    for(int i=0; i < v->size; i++){
        printf("%d ", v->data[i]);
    }
    printf("\n");
}



int main(){

    int capacidade;
    int operacao;
    char leitor[10];
    int num = 0;

    scanf("%d", &capacidade);
    vector *p = new_vector(capacidade);

    scanf("%d", &operacao);

    for(int i=0; i<operacao; i++){
        type valor = 0;
        int pos = 0;
        int remoP = 0;

        scanf("%s", leitor);


        if(strcmp(leitor, "pop_back") == 0){
            vector_pop_back(p);
            vector_print(p);
        }

        if(strcmp(leitor, "pop_front") == 0){
            vector_pop_front(p);
            vector_print(p);
        }

        if(strcmp(leitor,"insert") == 0){
            scanf("%d %d", &valor, &pos);
            vector_insert(p, valor, pos);
            vector_print(p);
        }

        if(strcmp(leitor,"push_back") == 0){
            scanf("%d", &valor);
            vector_push_back(p, valor);
            vector_print(p);
        }

        if(strcmp(leitor,"push_front") == 0){
            scanf("%d", &valor);
            vector_push_front(p, valor);
            vector_print(p);
        }

        if(strcmp(leitor,"set") == 0){
            scanf("%d %d", &pos, &valor);
            vector_set(p, pos, valor);
            vector_print(p);
        }

        if(strcmp(leitor,"erase") == 0){
            scanf("%d", &pos);
            remoP = vector_erase(p, pos);
            vector_print(p);
        }

    }

    free_vector(p);
}
