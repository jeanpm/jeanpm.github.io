vector* new_vector (int cap_inicial)
{
   if(cap_inicial==0)
    cap_inicial=4;
   vector* v =malloc(sizeof(vector));
   v->size = 0;
   v->data = malloc(sizeof(Type)*cap_inicial);
   v->capacity = cap_inicial;
   return v;

}
