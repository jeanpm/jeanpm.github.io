#include <stdio.h>
#include <stdlib.h>
#define CAMINHO 0
#define MURO 1
#define VISITADO 3
#define SAIDA 9
#include "list.h"

int resolver_labirinto(list* stack, int m, int n, int matrix[m][n], int x, int y)
{

    printf("%d %d\n",x,y);
    if(matrix[x][y]==SAIDA)
    {
        matrix[x][y]=VISITADO;
        return 1;
    }
    else
    {
        matrix[x][y]=VISITADO;     

        if(matrix[x][y-1]!=MURO && matrix[x][y-1]!=VISITADO)
        {
            list_push_back(stack,x);
            list_push_back(stack,y-1);
        }

        if(matrix[x][y+1]!=MURO && matrix[x][y+1]!=VISITADO)
        {
            list_push_back(stack,x);
            list_push_back(stack,y+1);
        }

        if(matrix[x-1][y]!=MURO && matrix[x-1][y]!=VISITADO)
        {
            list_push_back(stack,x-1);
            list_push_back(stack,y);
        }

        if(matrix[x+1][y]!=MURO && matrix[x+1][y]!=VISITADO)
        {
            list_push_back(stack,x+1);
            list_push_back(stack,y);
        } 

        return 0;
    }
}

void pilha(list* stack, int* x, int* y)
{

	if(!list_empty(stack))
    {   
        *y=list_pop_back(stack);
        *x=list_pop_back(stack);
    }
}

void fila(list* stack, int* x, int* y)
{

	if(!list_empty(stack))
    {   
        *x=list_pop_front(stack);
        *y=list_pop_front(stack);
    }
}

int main()
{
    // criação de uma lista
    list* stack = new_list();
    int x,y;
    scanf("%d %d",&x,&y);
    int m,n;
    scanf("%d %d",&m,&n);
    int matrix[m][n];

    // alimentando a matriz com os valores
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            scanf("%d",&matrix[i][j]);
        }
    }

    // empilhando os valores da posição inicial
    list_push_back(stack,x);
    list_push_back(stack,y);

    int result=0;
    while(result!=1)
    {
        // retornando o valor da função resolver_labirinto
        result=resolver_labirinto(stack,m,n,matrix,x,y);
        // para mudar para fila, é só mudar para a função fila e vice-versa
        pilha(stack,&x,&y);

        if(result==1)
        {
            // definindo x e y como -1 para a finalização do programa
            x=-1;
            y=-1;
        }       
    }

    if(x<0)
    {
        // printando -1 -1 ao final da solução
        printf("%d %d\n",x,y);
    }

    // Free na stack / fila alocada
    free_list(stack);

}
