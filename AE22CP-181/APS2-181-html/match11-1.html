#include "list.h"

//Aloca memória para uma matriz m x n e retorna um ponteiro para essa matriz.
int** new_matriz(int l, int c) {
    int **p = (int **) malloc(sizeof(int *) * l);
    for(int i = 0; i < l; i++) {
        p[i] = (int *) malloc(sizeof(int) * c);

    }

    return(p);
}

//Libera a memória previamente alocada em new_matrix
void free_matriz(int** matriz, int l) {
    for(int i = 0; i < l; i++) {
        free(matriz[i]);
    }
    free(matriz);
}

// Imprime os elementos da matriz. A primeira linha impressa deve ser as dimensões 'l c'
void print_matriz(int** matriz, int l, int c, int x, int y) {

    printf("\n%d %d", x, y);
    printf("\n%d %d", l, c);
    for(int i = 0; i < l; i++) {
        printf("\n");
        for(int j = 0; j < c; j++) {
            printf("%d ",matriz[i][j]);
        }
    }
}

void print_caminho(list* cam1, list* cam2) {
    while(list_size(cam1) != 0){
        printf("%d %d\n", list_pop_front(cam1), list_pop_front(cam2));
    }
}

void expansao(int** matriz, int a, int b) {

    list* cam1 = new_list();
    list* cam2 = new_list();

    list* x = new_list();
    list* y = new_list();

    matriz[a][b] = 2;

    list_push_back(cam1, a);
    list_push_back(cam2, b);

    while(matriz[a][b] != 9) {
        if(matriz[a+1][b] == 9 || matriz[a-1][b] == 9 || matriz[a][b+1] == 9 || matriz[a][b-1] == 9) {
            if(matriz[a+1][b] == 9){
                list_push_back(x, a+1);
                list_push_back(y, b);
            }
            else if(matriz[a-1][b] == 9){
                list_push_back(x, a-1);
                list_push_back(y, b);
            }
            else if(matriz[a][b+1] == 9){
                list_push_back(x, a);
                list_push_back(y, b+1);
            }
            else{
                list_push_back(x, a);
                list_push_back(y, b-1);
            }
        }
        else {
            // baixo
            if(matriz[a+1][b] == 0) {
                list_push_back(x, a+1);
                list_push_back(y, b);
                matriz[a][b] = 2;
            }
            //cima
            if(matriz[a-1][b] == 0) {
                list_push_back(x, a-1);
                list_push_back(y, b);
                matriz[a][b] = 2;
            }
            //direita
            if(matriz[a][b+1] == 0) {
                list_push_back(x, a);
                list_push_back(y, b+1);
                matriz[a][b] = 2;
            }
            //esquerda
            if(matriz[a][b-1] == 0) {
                list_push_back(x, a);
                list_push_back(y, b-1);
                matriz[a][b] = 2;
            }
        }


        a = list_pop_front(x);
        b = list_pop_front(y);
        list_push_back(cam1, a);
        list_push_back(cam2, b);
    }

    list_push_back(cam1, -1);
    list_push_back(cam2, -1);
    print_caminho(cam1, cam2);

    free_list(x);
    free_list(y);
    free_list(cam1);
    free_list(cam2);
}
