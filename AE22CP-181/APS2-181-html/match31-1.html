int** new_matrix(int m, int n)
{
    int** matrix;
    int i,j;
    matrix=malloc(sizeof(int*)*m);
    for(i=0;i<m;i++)
    {
        matrix[i]= malloc(sizeof(int)*n);
    }
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++)
        {
           matrix[i][j]=0;
        }
    }
    return matrix;
}
void free_matrix(int** matrix, int m)
{
    for(int i=0;i<m;i++)
    {
        free(matrix[i]);
    }
}
void print_matrix(int** matrix, int m, int n)
{
    printf("%d %d\n",m,n);
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            printf("%d ",matrix[i][j]);
        }
        printf("\n");
    }
}
void labirinto(int** matrix, int Xinicial, int Yinicial, int m, int n)
{
    int x = Xinicial;
    int y = Yinicial;
    list* v = new_list();
    printf("%d %d\n",x,y);
    while(matrix[x][y]!=exit)
    {
           matrix[x][y]=pass;
           if(x+1<m)
           {
               if(matrix[x+1][y]==path||matrix[x+1][y]==exit)
               {
                   list_push_back(v,x+1);
                   list_push_back(v,y);
               }
           }
           if(x-1>=0)
           {
               if(matrix[x-1][y]==path||matrix[x-1][y]==exit)
               {
                   list_push_back(v,x-1);
                   list_push_back(v,y);
               }
           }
           if(y+1<n)
           {
               if(matrix[x][y+1]==path||matrix[x][y+1]==exit)
               {
                   list_push_back(v,x);
                   list_push_back(v,y+1);
               }
           }
           if(y-1>=0)
           {
               if(matrix[x][y-1]==path||matrix[x][y-1]==exit)
               {
                   list_push_back(v,x);
                   list_push_back(v,y-1);
               }
           }
           //Pilha
