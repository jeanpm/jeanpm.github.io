int ** new_matrix(int m, int n)
{
	int **matrix = malloc(sizeof(int*)*m);
	for(int i=0; i<m;i++)
	{
		matrix[i] = malloc(sizeof(int)*n);
	}
	return matrix;
}
void free_matrix(int ** matrix,int m,int n)
{
	for(int i = 0; i< m;i++)
	{
		free(matrix[i]);
	}
	free(matrix);
}
void olhar(int** matrix,list* v,int pos_i,int pos_j)
{
	int c1 = 0,c2 = 0,c3 = 0,c4 = 0;
	int i = pos_i;
	int j = pos_j;
	//Se um caminho for percorrível, ele vai entrar na fila.
	if(matrix[i-1][j] == 0 || matrix[i-1][j] == 9)
	{
		c1 = 1;
	}
	if(matrix[i][j+1] == 0 || matrix[i][j+1] == 9)
	{
		c2 = 1;
	}
	if(matrix[i+1][j] == 0|| matrix[i+1][j] == 9)
	{
		c3 = 1;
	}
	if(matrix[i][j-1] == 0|| matrix[i][j-1] == 9)
	{
		c4 = 1;
	}
	//Primeiro insere na fila a coordenada do caminho percorrível em i(linha), e depois na coordenada j(coluna) no caso de fila.
	if(c1)
	{
		list_push_front(v,i-1);
		list_push_front(v,j);
	}
	if(c2)
	{
		list_push_front(v,i);
		list_push_front(v,j+1);
	}
	if(c3)
	{
		list_push_front(v,i+1);
		list_push_front(v,j);
	}
	if(c4)
	{
		list_push_front(v,i);
		list_push_front(v,j-1);
	}
}
int caminhar_i(list* v)
{
	return list_pop_back(v);
}
int caminhar_j(list* v)
{
	return list_pop_back(v);
}
