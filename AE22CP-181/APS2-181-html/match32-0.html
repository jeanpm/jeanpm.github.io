#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <stdbool.h>
#include "list.h"

int** new_matrix(int m, int n)
{
	int i;
	
	int** matrix= malloc(sizeof(int*)*m);

	for(i=0;i<m;i++)
	{
		matrix[i]= malloc(sizeof(int)*n);

	}

	return(matrix);
}

void solve_maze(int** matrix, int a, int b){

	list* v = new_list();

	while(matrix[a][b] != 9)
	{
		matrix[a][b]=2;

		if(matrix[a+1][b]!= 1 && matrix[a+1][b]!= matrix[a][b])
		{
			list_push_back(v, a+1);
			list_push_back(v, b);		
		}
		if(matrix[a-1][b]!= 1 && matrix[a-1][b]!= matrix[a][b])
		{
			list_push_back(v, a-1);
			list_push_back(v, b);
		}
		if(matrix[a][b+1]!= 1 && matrix[a][b+1]!= matrix[a][b])
		{
			list_push_back(v, a);
			list_push_back(v, b+1);
		}
		if(matrix[a][b-1]!= 1 && matrix[a][b-1]!= matrix[a][b])
		{
			list_push_back(v, a);
			list_push_back(v, b-1);
		}

		printf("%3d %3d", a, b);
		printf("\n");

		if(!list_empty(v))
		{
			b=list_pop_back(v);
			a=list_pop_back(v);	
		}
	}
}

void free_matrix(int** matrix)
{
	free(matrix);
	matrix=NULL;
}

int main(){

	int a, b;
	
	scanf("%d%d", &a, &b);

	int linha, coluna;

	scanf("%d%d", &linha, &coluna);

	int** matrix= new_matrix(linha, coluna);

	int i, j;
	for(i=0;i<linha;i++)
	{
		for(j=0;j<coluna;j++)
			scanf("%d", &matrix[i][j]);
	}

	solve_maze(matrix, a, b);
	free_matrix(matrix);
	a=-1;
	b=-1;

	printf("%3d %3d", a, b);
	printf("\n");

	return 0;
}
