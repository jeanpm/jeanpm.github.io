#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include "list.h"
#define EXIT 9
#define VISITEDWALL 1

int** new_matrix(int m, int n)
{
    int** matrix=malloc(sizeof(int*)*m);
    for(int i=0;i<m;i++)
    {
        matrix[i]=malloc(sizeof(int)*n);
    }

    return matrix;
}

void free_matrix(int** matrix, int m)
{
    for(int i=0;i<m;i++)
    {
        free(matrix[i]);
    }

    free(matrix);
}

int solve_maze(int** matrix, int x, int y, list* s)
{
    printf("%d %d\n",x,y);

    // Tanto faz definir visitado como muro ou outro valor
    if(matrix[x][y]==EXIT)
    {
        matrix[x][y]=VISITEDWALL;
        return 0;
    }
    else
    {
        matrix[x][y]=VISITEDWALL;

        if(matrix[x][y-1]!=VISITEDWALL)
        {
            list_push_back(s, x);
            list_push_back(s, y-1);
        }
        if(matrix[x][y+1]!=VISITEDWALL)
        {
            list_push_back(s, x);
            list_push_back(s, y+1);
        }
        if(matrix[x-1][y]!=VISITEDWALL)
        {
            list_push_back(s, x-1);
            list_push_back(s, y);
        }
        if(matrix[x+1][y]!=VISITEDWALL)
        {
            list_push_back(s, x+1);
            list_push_back(s, y);
        }

        return 999;
    }
}

int main()
{
    int x,y;
    scanf("%d %d", &x, &y);
    int m, n;
    scanf("%d %d", &m, &n);

    int** matrix = new_matrix(m, n);

    for (int i=0; i<m; i++)
    {
        for (int j=0; j<n; j++)
        {
            scanf("%d", &matrix[i][j]);
        }
    }

    list* s=new_list();

    int saida=999;

    list_push_back(s, x);
    list_push_back(s, y);

    while(saida!=0)
    {
        saida=solve_maze(matrix, x, y, s);

        if(!list_empty(s))
        {
            // Para mudar de fila para pilha ? s? mudar de front para back
            x=list_pop_front(s);
            y=list_pop_front(s);
        }
    }

    if(saida==0)
    {
        printf("-1 -1");
    }

    // free
    free_matrix(matrix, m);
    free_list(s);

    return 0;
}
