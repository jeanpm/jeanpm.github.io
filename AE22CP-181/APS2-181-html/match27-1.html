int **new_matrix(int m, int n) {
    int **matriz;
    int i;
    if(m<1||n<1)
    {
        printf("erro");
        return NULL;

    }
    matriz =(int**)malloc(m* sizeof(int*));
    if(matriz==NULL)
    {
        printf("memoria insuficiente");
        return NULL;
    }
    for (i = 0; i < m; i++)
    {
        matriz[i] = (int*)malloc(n * sizeof(int));
        if(matriz[i]==NULL)
        {
            printf("memoria insuficiente");
            return NULL;
        }
    }
    return matriz;
}

void  print_matrix(int** matrix, int m, int n) {
    int i, j;
    for (i = 0; i < m; i++) {
        for (j = 0; j < n; j++)
            printf("%d ", matrix[i][j]);
        printf("\n");
    }
}
