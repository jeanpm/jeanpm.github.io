#include<stdio.h>
#include<stdlib.h>
#include<string.h>

typedef int type;

typedef struct {

int* data;
int size;
int capacity;

}vector;

//Aloca memoria inicial para o vetor
vector* new_vector(int cap_ini)
{
vector* ve = malloc(sizeof(vector));
	ve->data = malloc(sizeof(int)*cap_ini);
	ve->capacity = cap_ini;
	ve->size = 0;
	return ve;
		
}
// Desaloca a memória de v->data e do próprio v.
void free_vector(vector* v)
{
	free(v->data);
	free(v);
}
// Retorna um ponteiro para o item na posição i
int* vector_at(vector* v, int i)
{
	return &(v->data[i]);

}
// Retorna o valor na posição i
int  vector_get(vector* v, int i)
{
	return (v->data[i]);
}

void resize(vector* v)
{
	if(v->size == v->capacity)
	{
		v->capacity = v->capacity*2;
		v->capacity = v->capacity;
		v->data = realloc(v->data,sizeof(int)*v->capacity);	
	}	
}
// Altera o valor na posição i
void  vector_set(vector* v, int value, int i)
{

	if( i< v->size)
	{
		v->data[i] = value;
	}
}

void vector_push_back(vector* v, int value)
{
	resize(v);
	v->data[v->size]=value;
	v->size++;	
	
}
void insert(vector* v, int value, int i)
{

	if (i >= v->capacity || v->size + 1 > v->capacity)
		resize(v);

	int k;
	for (k = v->size; k > i; k--)
		v->data[k] = v->data[k - 1];

	v->data[i] = value;
	v->size++;
	
}


 void vector_push_front(vector* v,int value)
{
 insert(v,value,0);
}
void vector_pop_back(vector* v)
{
  	v->size--; 
	
}

type erase (vector* v, int i)
{
    	int k;
	for (k = i + 1; k < v->size; k++)
		v->data[k - 1] = v->data[k];

	v->size--;
	return 1;

}
