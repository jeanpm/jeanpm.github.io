#include <stdlib.h>
#include <stdio.h>
#include <assert.h>
#include <string.h>

typedef int Type;
typedef struct {
	Type *data;
	int size;
	int capacity;
} vector;

vector *new_vector(int initial_capacity) {
	vector *v = (vector *) malloc(sizeof(vector));

	if(initial_capacity > 0)
		v->capacity = initial_capacity;
	else
		v->capacity = 4;

	v->size = 0;
	v->data = (Type *) malloc(sizeof(Type)*v->capacity);
	
	return v;
}

void free_vector(vector *v) {
	free(v->data);
	free(v);
}

void set(vector *v, int index, Type value) {
	assert(index >= 0 && index < v->size);
	v->data[index] = value;
}

void resize(vector *v) {
	v->capacity *= 2;
	v->data = (Type *) realloc(v->data, sizeof(Type)*v->capacity);
}

void insert(vector *v, int value, int index) {
	assert(index >= 0 && index <= v->size);
	if(v->size == v->capacity)
		resize(v);

	for(int i = v->size; i > index; i--)
		v->data[i]= v->data[i-1];

	v->data[index] = value;
	v->size++;
}

void pushfront(vector *v, Type value) {
	insert(v, value, 0);
}

void pushback(vector *v, Type value) {
	insert(v, value, v->size);
}

void erase(vector *v, int index) {
	assert(index >= 0 && index < v->size);
	v->size--;
	for(int i = index+1; i <= v->size; i++)
		v->data[i-1] = v->data[i];
}

void popfront(vector *v) {
	erase(v, 0);
}

void popback(vector *v) {
	erase(v, v->size-1);
}

void print_vector(vector *v) {
	printf("[%d/%d]", v->size, v->capacity);
	for(int i = 0; i < v->size; i++)
		printf(" %d", v->data[i]);
	printf("\n");
}

int main() {
	int start, lines;
	scanf("%d", &start);
	scanf("%d", &lines);
	
	vector *thing = new_vector(start);
	char command[20];
	int value, index;

	for(int i = 0; i < lines; i++) {
		scanf(" %s", command);
		
		if(strcmp(command, "push_back") == 0) {
			scanf("%d", &value);
			pushback(thing, value);

		} else if(strcmp(command, "push_front") == 0) {
			scanf("%d", &value);
			pushfront(thing, value);

		} else if(strcmp(command, "insert") == 0) {
			scanf("%d%d", &value, &index);
			insert(thing, value, index);

		} else if(strcmp(command, "pop_back") == 0) {
			popback(thing);

		} else if(strcmp(command, "pop_front") == 0) {
			popfront(thing);

		} else if(strcmp(command, "erase") == 0) {
			scanf("%d", &index);
			erase(thing, index);

		} else if(strcmp(command, "set") == 0) {
			scanf("%d %d", &index, &value);
			set(thing, index, value);
		}
		print_vector(thing);
	}
	free_vector(thing);
	return 0;
}
