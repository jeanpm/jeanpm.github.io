#include <stdio.h>
#include <stdlib.h>
#include <string.h>


typedef int Type;

typedef struct {
	Type* data; //vetor que armazenará o inteiro
	int size;  //tamanho atual do vetor
	int capacity; //tamanho reservado em memória
}vector;

// Aloca memória inicial para o vetor
//inicialização
vector *new_vector(int initial_capacity) {
	vector *v = (vector*)malloc(sizeof(vector));
	v -> data = (Type*)(malloc(sizeof(int)*initial_capacity));
	v -> size = 0;
	v -> capacity = initial_capacity;
	return v;
}

//finalização
void free_vector(vector *v) {
	free(v -> data);
	free(v);
}

// Altera o valor na posição i
void vector_set(vector *v, int i, Type value) {
	v -> data[i] = value;
}

void resize(vector *v) {
	if(v -> capacity == 0)
		v -> capacity = 4;
	else 
		v -> capacity = v -> capacity * 2;

	v -> data = (Type*)(realloc(v->data, sizeof(int)*v->capacity));
}

//acessso e manipulação: com crescimento
void vector_push_back(vector *v, Type value) {	
	if(v -> size == v -> capacity) 
		resize(v); 

	v -> data[v -> size++] = value;	
}

void vector_insert(vector* v, Type value, int i) {
	int j;
	
	if(v -> size == v -> capacity) 
		resize(v); 

	for (j = v -> size; j > i; j--)
		v -> data[j] = v -> data[j-1];

	v -> size += 1;
	v -> data[i] = value;
}

void vector_push_front(vector *v, Type value) {
	vector_insert(v, value, 0);
}	

void vector_print(vector* v) {
	int i;
	printf("[%d/%d] ", v -> size, v -> capacity);
	for(i = 0; i < v -> size; i++) 
		printf("%d ", v -> data[i]);
	printf("\n");

}

void vector_pop_back(vector* v){
	if(v -> size > 0)
		v -> size = v -> size - 1;
}

void vector_pop_front(vector* v) {
	int i;

	for (i = 0; i < v -> size; i++) 
	v -> data[i] = v -> data[i+1];

	if(v -> size > 0)
		v -> size = v -> size - 1;
}	

void vector_erase(vector* v, int i) {
	int j;

	if(v->size > 0) {
		for (j = i; j < v -> size; j++) 
			v -> data[j] = v -> data[j+1];

		v -> size -= 1;
	}
}

int main () {
	char comando[20];
	vector *v;
	int init, nLinhas=0, num, i, indice, j;
	
	scanf(" %d", &init);

	if(init == 0)
		init = 4;

	v = new_vector(init);

	scanf("%d", &nLinhas);


	for(i=0; i < nLinhas; i++) {
		scanf(" %s", comando);

		if(strcmp(comando, "push_back") == 0) {
			scanf(" %d", &num);
			vector_push_back(v, num);
		}
		else if (strcmp(comando, "set") == 0){
			scanf("%d%d", &indice, &num);
			vector_set(v, indice, num);
		}
		else if(strcmp(comando, "push_front") == 0) {
			scanf("%d", &num);
			vector_push_front(v, num);
		}
		else if(strcmp(comando, "insert") == 0) {
			scanf("%d%d", &num, &indice);
			vector_insert(v, num, indice);
		}
		else if(strcmp(comando, "pop_back") == 0)
			vector_pop_back(v);

		else if(strcmp(comando, "pop_front") == 0)
			vector_pop_front(v);

		else if (strcmp(comando, "erase") == 0) {
			scanf("%d", &indice);
			vector_erase(v, indice);
		}
		vector_print(v);
	}
	free_vector(v);
	return 0;
}
