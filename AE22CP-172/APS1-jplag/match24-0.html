#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<assert.h>


typedef int Type;

typedef struct{
    Type* data;
    int size;
    int capacity;
} vector;

vector* new_vector(int init_cap)
{
    if(init_cap<=0)
        init_cap=4;

    vector* b=malloc(sizeof(vector));
    b->data=malloc(sizeof(int)*init_cap);
    b->capacity=init_cap;
    b->size=0;

    return(b);
}

void free_vector(vector *v)
{
    free(v->data);
    free(v);
}

void resize(vector* v)
{
    v->capacity+=v->capacity;
    v->data=realloc(v->data, sizeof(int)*v->capacity);
}

Type* vector_at(vector* v, int i)
{
    return (&((v->data)[i]));
}

Type vector_get(vector* v, int i)
{
    return ((v->data)[i]);
}

void vector_set(vector* v, Type value, int i)
{
    if(v->size > i)
        (v->data)[i]=value;
}

void vector_push_back(vector *v, Type value)
{
    if(v->size >= v->capacity)
        resize(v);

    (v->data)[v->size]=value;
    v->size++;
}

void vector_insert(vector* v, Type value, int i)
{
    if(v->size == v->capacity)
        resize(v);

    int j;
    for(j=(v->size); j >=i ; j--)
        (v->data)[j]=(v->data)[j-1];
    v->size++;
    (v->data)[i]=value;
}

void vector_push_front(vector* v, Type value)
{
    vector_insert(v, value, 0);
}

void vector_pop_back(vector *v)
{
    v->size--;
}

void vector_erase(vector *v, int i)
{
    int j;

    for(j=i; j<(v->size-1); j++)
        v->data[j] = v->data[j+1];
    vector_pop_back(v);
}

void vector_pop_front(vector *v)
{
    vector_erase(v, 0);
}

void vector_print(vector *v)
{
    printf("[%i/%i] ", v->size, v->capacity);

    int i;
    for(i=0; i<v->size; i++)
        printf("%d ", v->data[i]);
    printf("\n");
}



int main()
{
    int cap, quant, i, par1, par2;
    char teste[20];

    scanf("%i%i", &cap, &quant);
    vector *v=new_vector(cap);

    for(i=0; i<quant; i++)
    {
        scanf("%s", teste);
        if(strcmp(teste, "push_back")==0)
        {
            scanf("%i", &par1);
            vector_push_back(v, par1);
        }
        else if(strcmp(teste, "push_front")==0)
        {
            scanf("%i", &par1);
            vector_push_front(v, par1);
        }
        else if(strcmp(teste, "insert")==0)
        {
            scanf("%i%i", &par1, &par2);
            vector_insert(v, par1, par2);
        }
        else if(strcmp(teste, "pop_back")==0)
        {
            vector_pop_back(v);
        }
        else if(strcmp(teste, "pop_front")==0)
        {
            vector_pop_front(v);
        }
        else if(strcmp(teste, "erase")==0)
        {
            scanf("%i", &par1);
            vector_erase(v, par1);
        }
        else if(strcmp(teste, "set")==0)
        {
            scanf("%i%i", &par1, &par2);
            vector_set(v, par2, par1);
        }
        vector_print(v);
    }
    return 0;
}
