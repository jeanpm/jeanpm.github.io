#include <string.h>
#include <time.h>
#include <assert.h>
#include <stdio.h>
#include <stdlib.h>


typedef int Type;

typedef struct
{

	int *data;
	int size;
	int capacity;

} vector;

void insert(vector *v, Type value, int i);
Type erase(vector *v, int i);

vector *new_vector(int intial_capacity)
{
	vector *v = malloc(sizeof(vector));
	v->data = malloc(sizeof(int) * intial_capacity);
	v->size = 0;
	v->capacity = intial_capacity;

	return v;
}

Type pop_front(vector *v)
{
	erase(v, 0);
	return 1;
}
Type pop_back(vector *v)
{
	v->size--;
	return 1;
}
void free_vector(vector *v)
{
	free(v->data);
	free(v);
}


Type erase(vector *v, int i)
{
	int j;
	for (j = i + 1; j < v->size; j++)
		v->data[j - 1] = v->data[j];

	v->size--;
	return 1;
}


int get(vector *v, int i)
{
	return v->data[i];
}

int *at(vector *v, int i)
{
	return &(v->data[i]);
}

void set(vector *v, Type value, int i)
{
	if (i < v->size)
		v->data[i] = value;
}

void resize(vector *v)
{
	if (v->capacity == 0)
		v->capacity = 4;

	else
		v->capacity *= 2;

	v->data = realloc(v->data, sizeof(Type) * v->capacity);
}

void push_back(vector *v, Type value)
{
	insert(v, value, v->size);
}

void insert(vector *v, Type value, int i)
{
	if (i >= v->capacity || v->size + 1 > v->capacity)
		resize(v);

	int j;
	for (j = v->size; j > i; j--)
		v->data[j] = v->data[j - 1];

	v->data[i] = value;
	v->size++;
}

void push_front(vector *v, Type value)
{
	insert(v, value, 0);
}

void print(vector *v)
{
	int i;

	printf("[%d/%d]", v->size, v->capacity);
	for (i = 0; i < v->size; i++)
		printf(" %d", v->data[i]);
	printf("\n");
}

int main(int argc, char *argv[])
{
	int n, i, valor, intial_capacity = 5, cap_inicial;
    char funcao[20];

	scanf("%d", &intial_capacity);
	vector *v = intial_capacity != 0 ? new_vector(intial_capacity) : new_vector(4);

	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		scanf("%s", funcao);


		if (strcmp(funcao, "push_front") == 0)
		{
			scanf("%d", &cap_inicial);
			push_front(v, cap_inicial);
		}
        if (strcmp(funcao, "push_back") == 0)
		{
			scanf("%d", &cap_inicial);
			push_back(v, cap_inicial);
		}
		else if (strcmp(funcao, "insert") == 0)
		{
			scanf("%d %d", &cap_inicial, &valor);
			insert(v, cap_inicial, valor);
		}

		else if (strcmp(funcao, "erase") == 0)
		{
			scanf("%d", &cap_inicial);
			erase(v, cap_inicial);
		}

        else if (strcmp(funcao, "set") == 0)
		{
			scanf("%d %d", &valor, &cap_inicial);
			set(v, cap_inicial, valor);
		}
		else if (strcmp(funcao, "pop_front") == 0)
			pop_front(v);

		else if (strcmp(funcao, "pop_back") == 0)
			pop_back(v);

		print(v);
	}

	free_vector(v);

	return 0;
}
