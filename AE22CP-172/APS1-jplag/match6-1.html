#include <stdlib.h>
#include <stdio.h>
#include <string.h>

typedef int Type;

typedef struct
{
    int *data;
    int size;
    int capacity;
} vector;

//Basics
vector *new_vector(int initial_capacity);
void free_vector(vector *v);
void resize(vector *v);
//Tools
void vector_print(vector *v);
void valid_id(vector *v, int i);
//Get, Set and At
Type *vector_at(vector *v, int i);
Type vector_get(vector *v, int i);
void vector_set(vector *v, int value, int i);
//Insert
void vector_insert(vector *v, int value, int i);
void vector_push_back(vector *v, int value);
void vector_push_front(vector *v, int value);
//Remove
Type vector_pop_back(vector *v);
Type vector_erase(vector *v, int i);
Type vector_pop_front(vector *v);

vector *new_vector(int initial_capacity)
{
    vector *v = malloc(sizeof(vector));
    v->data = malloc(initial_capacity * sizeof(int));
    v->size = 0;
    v->capacity = initial_capacity;
    return v;
}
void free_vector(vector *v)
{
    free(v->data);
    free(v);
}
void resize(vector *v)
{
    int new_capacity = v->capacity * 2;
    v->data = realloc(v->data, sizeof(int) * new_capacity);
    if (v->data)
    {
        for (int i = v->capacity; i < new_capacity; i++)
            v->data[i] = '\0';
        v->capacity = new_capacity;
    }
    else
    {
        printf("Memoria insuficiente\n");
        exit(1);
    }
}
void vector_print(vector *v)
{
    printf("[%d/%d]", v->size, v->capacity);
    for (int i = 0; i < v->size; i++)
    {
        printf("%d ", vector_get(v, i));
    }
    printf("\n");
}
void valid_id(vector *v, int i)
{
    if (i >= 0 && i >= v->size)
    {
        printf("\nErro: i=%d, size=%d\n", i, v->size);
        exit(1);
    }
}
Type *vector_at(vector *v, int i)
{
    valid_id(v, i);
    return &(v->data[i]);
}
Type vector_get(vector *v, int i)
{
    valid_id(v, i);
    return v->data[i];
}
void vector_set(vector *v, int value, int i)
{
    valid_id(v, i);
    v->data[i] = value;
}

void vector_insert(vector *v, int value, int i)
{
    while (v->capacity <= i || v->capacity <= v->size)
        resize(v);

    if (i < v->size)
    {
        v->size++;
        for(int j = (v->size - 1); j > i; j--)
        {
            vector_set(v, vector_get(v,j-1), j);
        }
    }
    else
        v->size = i + 1;

    vector_set(v, value, i);
}
void vector_push_back(vector *v, int value)
{
    vector_insert(v, value, v->size);
}
void vector_push_front(vector *v, int value)
{
    vector_insert(v, value, 0);
}

Type vector_erase(vector *v, int i)
{
    if (i == (v->size - 1))
        vector_pop_back(v);
    else
    { 
        for(int j = i; j < (v->size - 1); j++)
        {
            vector_set(v, vector_get(v,j+1), j);
        }   
        v->size--;     
    }
    return 1;
}
Type vector_pop_back(vector *v)
{
    return v->data[v->size--];
}

int vector_pop_front(vector *v)
{
    return vector_erase(v,0);
}

int main(int argc, char *argv[])
{
    char * out = "";
    char cmd[20];
    int p1 = 0, p2, count_c;

    scanf("%d",&p1);
    vector *V = p1 ? new_vector(p1) : new_vector(4);

    scanf("%d",&count_c);
    for (int i = 0; i < count_c; i++)
    {
        scanf("%s", cmd);
        if (strcmp(cmd, "push_back") == 0)
        {
            scanf("%d", &p1);
            vector_push_back(V, p1);
        }

        if (strcmp(cmd, "push_front") == 0)
        {
            scanf("%d", &p1);
            vector_push_front(V, p1);
        }
        else if (strcmp(cmd, "insert") == 0)
        {
            scanf("%d %d", &p1, &p2);
            vector_insert(V, p1, p2);
        }
        else if (strcmp(cmd, "set") == 0)
        {
            scanf("%d %d", &p2, &p1);
            vector_set(V, p1, p2);
        }
        else if (strcmp(cmd, "erase") == 0)
        {
            scanf("%d", &p1);
            vector_erase(V, p1);
        }
        else if (strcmp(cmd, "pop_back") == 0)
            vector_pop_back(V);

        else if (strcmp(cmd, "pop_front") == 0)
            vector_pop_front(V);

        vector_print(V);
    }

    free_vector(V);
    
    return 0;
}
