#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <string.h>

typedef int type;
typedef struct {
    type* data;
    int size;
    int capacity;
} vector;
vector* new_vector(int initial_capacity) {
    vector *v = (vector*)malloc(sizeof(vector));
    v -> data = (type*)malloc(sizeof(int)*initial_capacity);
    v -> size = 0;
    v -> capacity = initial_capacity;
    return v;
}

//finaliza??o
void vector_free(vector* v) {
    free(v -> data);
    free(v);
}

void resize(vector* v) {
    if(v->capacity==0)
    {
        v->data = (type*)realloc(v->data, sizeof(int)*(v->capacity+4));
        v->capacity = v->capacity+4;
    }
    else if(v->size==v->capacity)
    {
        v->data = (type*)realloc(v->data, sizeof(int)*(v->capacity*2));
        v->capacity = v->capacity*2;
    }
}

void vector_push_back(vector* v, int value) {
    resize(v);
    v->data[v->size]=value;
    v->size++;
}

void vector_push_front(vector* v, int value) {
    int a;
    resize(v);
    for(a=v->size; a>0; a--)
    {
        v->data[a]=v->data[a-1];
    }
    v->size++;
    v->data[0]=value;
}

void vector_pop_back(vector* v) {
    if(v->size>0)
    v->size--;
}

void vector_pop_front(vector* v) {
    int i;
    for(i=0; i<v->size; i++)
    {
        v->data[i]=v->data[i+1];
    }
    v->size--;
}

void vector_erase(vector* v, int i) {
    for(i; i<v->size; i++)
    {
        v->data[i]=v->data[i+1];
    }
    v->size--;
}

void vector_insert(vector* v, int value, int i) {
    int a;
    resize(v);
    for(a=v->size; a>i; a--)
    {
        v->data[a]=v->data[a-1];
    }
    v->size++;
    v->data[i]=value;
}

void vector_print(vector* v) {
    int i;
    printf("[%d/%d]", v->size, v->capacity);
    for(i=0; i<v->size; i++)
    {
        printf(" %d", v->data[i]);
    }
    printf("\n");
}

void vector_set(vector* v, int pos, int value) {
    v->data[pos]=value;
}
