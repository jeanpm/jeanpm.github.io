typedef struct
{
    int* data;  // vetor com os inteiros
    int size;   // tamanho atual do vetor
    int capacity;   // tamanho reservado em memoria(max)
} vector;

vector* new_vector(int inicial_capacity)
{
    vector* v= malloc(sizeof(vector*));
    v->size=0;
    v->capacity= inicial_capacity;
    v->data= malloc(sizeof(int)*inicial_capacity);
    return v;
}

void resize(vector* v)
{
    if(v->capacity==0)
    {
        v->capacity=4;
    }
    else
    {
    v->capacity=2*(v->capacity);
    v->data=realloc(v->data, sizeof(int)*v->capacity);
    }
}


void free_vector(vector* v)
{
    free(v->data);
    free(v);
}

void vector_push_back(vector* v, int value)
{
    if(v->capacity==v->size)
    {
        v->capacity*=2;
        v->data=realloc(v->data, sizeof(int)* v->capacity);//
    }
    v->data[v->size]= value;
    v->size+=1;
}



void vector_insert(vector* v, int value, int i)
{
    if(v->size==v->capacity)
        resize(v);
    if(i>v->size)
        printf("error: not aloccated ");

    else
    {
        for(int j= v->size;;j--)
        {
            if(j==i)
                break;
            v->data[j]=v->data[j-1];
            
        }
        v->size++;
        v->data[i]= value;
    }
}

void vector_set(vector* v, int value, int i)
{
    if(i>v->size)
        printf("error: not aloccated ");
    else
    {
        v->data[i]= value;
    }
    
}

void vector_push_front(vector* v, int i)
{
    if(v->capacity==v->size)
    {
        resize(v);
    }
    for(int j=v->size-1;j>=0;j--)
    {
        v->data[j+1]=v->data[j];
    }
    v->size++;
    v->data[0]=i;
}

void vector_pop_back(vector* v)
{
    v->size--;
    v->data[v->size]=0;
}

void vector_erase(vector* v, int i)
{
    if(i>=v->size)
        printf("error");
    else
    {
        v->data[i]=0;
        for(int j=i;;j++)
        {
            if(j==v->size-1)
                break;
            v->data[j]=v->data[j+1];
        }
    }
    v->data[v->size]=0;
    v->size--;
}

void vector_pop_front(vector* v)
{
    v->data[0]=0;
    v->size--;
    for(int i=0;i<v->size;i++)
    {
        v->data[i]=v->data[i+1];
    }
}

void vector_print(vector* v)
{
    printf("[%d/%d] ",v->size, v->capacity );
    for(int i=0;i<v->size;i++)
    {
        printf("%d ",v->data[i]);
    }
    printf("\n");
}
